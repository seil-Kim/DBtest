	<script language = "javascript">
		function check(){
			var blank = 0;
			if(form1.id.value == "") blank = 1;
			if(form1.password.value == "") blank = 1;
			if(form1.name.value == "") blank = 1;
			if(form1.email.value == "") blank = 1;
			if(blank == 1){
				alert("모든 내용을 입력해 주세요.");
				return(false)
			}
			else return(true);
		}
		function check_num(str){
			if(isNaN(str)){
				alert("숫자만 입력하세요.");
			return "";
			}else return str;
		}
	</script>
	<form action = insert.jsp method = post name = form1 onsubmit = "return check();">
		사용자 ID <input type = "text" name = "id"><br>
		암호 <input type = "password" name = "password" onkeyup = "value = check_num(value)"><br>

		E-mail <input type = "text" name = "email"><br>
